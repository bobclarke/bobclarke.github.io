#!/bin/bash
for i in "$@"
do
case $i in
    -p=*|--prefix=*)
    PREFIX="${i#*=}"
    ;;
 
    -s=*|--searchpath=*)
    SEARCHPATH="${i#*=}"
    ;;
 
    -l=*|--lib=*)
    DIR="${i#*=}"
    ;;
 
    --default)
    DEFAULT=YES
    ;;
 
    *)
    ;;
esac
done
echo PREFIX = ${PREFIX}
echo SEARCH PATH = ${SEARCHPATH}
echo DIRS = ${DIR}
echo DEFAULT = ${DEFAULT}
